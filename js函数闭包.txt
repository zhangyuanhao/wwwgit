<script>
var add = (function () {
    var counter = 0;
	var flag = true;
    return function () {
		if (flag) {
			flag = false;
			return counter += 1;
		}
		else { 
			flag = true;
			return counter -=1;
		}
	}
})();
function myFunction(){
    document.getElementById("demo").innerHTML = add();
}
</script>
